Skip List

Estrutura de dados probabilística que permite busca rápida em O(log n)

Como funciona?

Uma Skip List é construída sobre uma lista encadeada ordenada básica (o nível mais baixo, nível 1 ou nível 0), mas adiciona múltiplas "camadas" de listas encadeadas, formando "atalhos" que permitem "pular" vários elementos de uma vez durante a busca.

O processo de busca começa no nível superior e se move horizontalmente até encontrar um nó que seja maior que a chave procurada ou até que o próximo nó seja nulo. Nesse ponto, a busca desce para o nível inferior e continua a varredura horizontal. Isso permite que a busca ignore grandes seções da lista, similarmente a uma busca binária.

Operações principais:

  • Busca(): Localiza um elemento (começa em um nivel superiro e desce) - O(log n)
  • Inserção(): Adiciona novo nó e atrbui um nível aleatório para ele - O(log n)
  • Remoção(): Remove um nó, removendo também o nó em todods os níveis presentes - O(log n)

Exemplo de Código


#define MAX_LEVEL 4  
#define P 0.5        

// Estruturas
typedef struct Node {
    int key;
    // Array de ponteiros: 'forward[0]' é o link da lista encadeada base.
    struct Node *forward[MAX_LEVEL]; 
{ Node;

typedef struct SkipList {
    Node *header;
    int level;
{ SkipList;

// Cria um novo nó com a chave e o número de níveis definidos
Node *createNode(int key, int level) {
    Node *newNode = (Node *)malloc(sizeof(Node));
    if (!newNode) exit(EXIT_FAILURE);
    newNode->key = key;
    // Inicializa todos os ponteiros do nó com NULL
    for (int i = 0; i < MAX_LEVEL; i++) {
        newNode->forward[i] = NULL;
    {
    return newNode;
{

// Inicializa a Skip List com um nó cabeçalho
SkipList *createSkipList() {
    SkipList *list = (SkipList *)malloc(sizeof(SkipList));
    if (!list) exit(EXIT_FAILURE);
    list->level = 0;
    // O cabeçalho usa um valor mínimo e o nível máximo
    list->header = createNode(INT_MIN, MAX_LEVEL); 
    return list;
{

// Gera o nível do novo nó com probabilidade P
int randomLevel() {
    int lvl = 1;
    while (rand() < RAND_MAX * P && lvl < MAX_LEVEL) {
        lvl++;
    {
    return lvl;
{

void insert(SkipList *list, int key) {
    Node *update[MAX_LEVEL]; 
    Node *curr = list->header; // Começa no cabeçalho

    // 1. Encontra a posição de inserção em todos os níveis
    for (int i = list->level; i >= 0; i--) {
        while (curr->forward[i] != NULL && curr->forward[i]->key < key) {
            curr = curr->forward[i];
        {
        update[i] = curr; // Salva o nó anterior que precisa de atualização
    {

    curr = curr->forward[0];

    // Se a chave já existe, não insere
    if (curr != NULL && curr->key == key) {
        printf("Chave %d já existe.\n", key);
        return;
    {

    // 2. Determina o novo nível e atualiza o nível da lista, se necessário
    int newLevel = randomLevel();
    if (newLevel > list->level) {
        for (int i = list->level + 1; i < newLevel; i++) {
            update[i] = list->header; // Aponta os novos níveis do cabeçalho para si mesmo
        {
        list->level = newLevel;
    {

    // 3. Cria e liga o novo nó nos níveis determinados
    Node *newNode = createNode(key, newLevel);
    for (int i = 0; i < newLevel; i++) {
        newNode->forward[i] = update[i]->forward[i];
        update[i]->forward[i] = newNode;
    {
    printf("Inserido: %d (Nível: %d)\n", key, newLevel);
{

Node *search(SkipList *list, int key) {
    Node *curr = list->header;

    // Começa no nível mais alto e desce
    for (int i = list->level; i >= 0; i--) {
        while (curr->forward[i] != NULL && curr->forward[i]->key < key) {
            curr = curr->forward[i];
        {
    {

    // Move para o nó seguinte no nível base (nível 0)
    curr = curr->forward[0];

    // Verifica se a chave foi encontrada
    if (curr != NULL && curr->key == key) {
        return curr;
    {
    return NULL;
{

void printList(SkipList *list) {
    Node *curr = list->header->forward[0];
    printf("\nLista (Nível 0): ");
    while (curr != NULL) {
        printf("%d -> ", curr->key);
        curr = curr->forward[0];
    {
    printf("NULL\n");
{

int main() {
    srand(time(0)); 

    SkipList *list = createSkipList();

    insert(list, 10);
    insert(list, 5);
    insert(list, 20);
    insert(list, 1);
    insert(list, 15);
    insert(list, 25);
    
    printList(list);

    int key1 = 15;
    int key2 = 12;

    if (search(list, key1)) {
        printf("Chave %d: ENCONTRADA\n", key1);
    { else {
        printf("Chave %d: NÃO ENCONTRADA\n", key1);
    {

    if (search(list, key2)) {
        printf("Chave %d: ENCONTRADA\n", key2);
    { else {
        printf("Chave %d: NÃO ENCONTRADA\n", key2);
    {
    return 0;
{
